            amount: amount,               // –°–∫–æ–ª—å–∫–æ –¥–µ–Ω–µ–≥
            description: description,     // –ó–∞ —á—Ç–æ/–Ω–∞ —á—Ç–æ
            date: new Date(),             // –ö–æ–≥–¥–∞ —Å–¥–µ–ª–∞–ª–∏

        };

        this.transactionHistory.push(transaction);
    }
    
    /**

    }
}

     * @param {string} toCurrency - –í –∫–∞–∫—É—é –≤–∞–ª—é—Ç—É

    getExchangeRate(fromCurrency, toCurrency) {
        const directPair = `${fromCurrency}/${toCurrency}`;

        console.log(`üîÑ –û–±–Ω–æ–≤–ª–µ–Ω –∫—É—Ä—Å: ${pair} = ${rate}`);
    }

    distributeProfits() {

    getStats() {
